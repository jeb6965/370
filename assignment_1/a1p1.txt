#include <iostream>
#include "CImg.h"
#include <cmath>
using namespace cimg_library;
int main(){
	//read in an image file that's in the same folder as your code
	CImg<unsigned char> image_1("nypl_left.bmp");
	//TODO: load second image for the anaglyph
	CImg<unsigned char> image_2("nypl_right.bmp");
	//get image dimensions
	std::cout << "image width: " << image_1.width() << std::endl; //how many columns are in the image
	std::cout << "image height: " << image_1.height() << std::endl; //how many rows are in the image
	//create an empty image
	//specify the width, height, Z-value (depth), and number of color channels (3 for RGB, 1 for grayscale)
	//TODO: create a offset variable 5 or 10 pixels to start
	int offset = 8;
	CImg<unsigned char> anaglyph(image_1.width(), image_1.height(), 1, 3, 0);
	CImg<unsigned char> red(image_1.width(), image_1.height(), 1, 3, 0);
	CImg<unsigned char> cyan(image_1.width(), image_1.height(), 1, 3, 0);
	//for (start, stop, step size)
	for (int x = 0; x < image_1.width(); x++){
		for (int y = 0; y < image_1.height(); y++){
			/*
			//to access the red value at pixel location (x,y): image_1(x, y,
			0, 0)
			//to access the green value at pixel location (x,y): image_1(x,
			y, 0, 1)
			//to access the blue value at pixel location (x,y): image_1(x, y,
			0, 2)
			*/
			//copy color from first image into second
			red(x, y, 0, 0) = image_1(x, y, 0, 0);
			red(x, y, 0, 1) = 0;
			red(x, y, 0, 2) = 0;

			cyan(x, y, 0, 0) = 0;
			cyan(x, y, 0, 1) = image_2(x, y, 0, 1);
			cyan(x, y, 0, 2) = image_2(x, y, 0, 2);

			anaglyph(x, y, 0, 0) = image_1(x, y, 0, 0);
			anaglyph(x + offset, y, 0, 1) = image_2(x, y, 0, 1);
			anaglyph(x + offset, y, 0, 2) = image_2(x, y, 0, 2);
			//TODO: copy red color channel from image 1, then green/blue from the second to form anaglyph.
			/*
			Notes
			compare your calculations to 0 and 255
			max(val1, val2)
			min(val1, val2)
			for invert subtract from 255, not 1
			for contrast subtract (256/3) or 85
			doing multiplication or division, use floats or doubles
			int value = 5;
			(1.0*value) -> results in a float.
			*/
		}
	}
	//display an image
	CImgDisplay disp(anaglyph);
	while (!disp.is_closed()) //can skip {} for 1 line blocks
	disp.wait();
	// used as needed, you may save the image before processing.
	// use a proper name based on the properties of the generated image.
	red.save("nypl_red.bmp");
	cyan.save("nypl_cyan.bmp");	
	anaglyph.save("nypl_anaglyph.bmp");
	return 0;
}
